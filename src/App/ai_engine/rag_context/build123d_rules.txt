---- EXEMPLO 2: Cilindro com fillet nas arestas superiores ----

from build123d import BuildPart, Cylinder, GeomType, fillet

with BuildPart() as part:
    # Cilindro base: R=12 mm, H=30 mm
    Cylinder(radius=12, height=30)

    # Arredondar arestas circulares do topo (Z positivo)
    top_z = 30 / 2          # = 15 (centro em Z=0, topo em Z=+15)
    top_edges = [
        e for e in part.edges()
        if abs(e.center().Z - top_z) < 0.5
    ]
    if top_edges:
        fillet(top_edges, radius=2.0)


---- EXEMPLO 3: Base + cilindro no topo com Locations ----

from build123d import BuildPart, Box, Cylinder, Locations, Location, fillet, GeomType

BASE_W, BASE_D, BASE_H = 60.0, 40.0, 10.0
CYL_R,  CYL_H         = 10.0, 25.0

with BuildPart() as part:
    # 1. Base retangular
    Box(length=BASE_W, width=BASE_D, height=BASE_H)

    # 2. Cilindro no topo: deslocar Z = BASE_H/2 + CYL_H/2
    with Locations(Location((0, 0, BASE_H / 2 + CYL_H / 2))):
        Cylinder(radius=CYL_R, height=CYL_H)

    # 3. Fillet nas arestas lineares superiores da base
    top_z = BASE_H / 2
    top_edges = [
        e for e in part.edges()
        if e.geom_type == GeomType.LINE and abs(e.center().Z - top_z) < 0.5
    ]
    if top_edges:
        fillet(top_edges, radius=2.0)
